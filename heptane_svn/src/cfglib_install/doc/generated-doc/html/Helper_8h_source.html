114 inline std::vector<std::string> split_string(
115 std::string const& input,
116 std::string const& delimiter,
117 bool empty_fields=false)
118 {
119 using std::vector ;
120 using std::string ;
121
122 vector<string> result ;
123
124 if ((input.size() == 0 ) || (delimiter.size() == 0))
125 { return result; }
126
127 string::size_type size_delim = delimiter.size() ;
128 string::size_type prec_pos = 0;
129 string::size_type cur_pos = input.find(delimiter, 0);
130 while (cur_pos != string::npos)
131 {
132 string cur(input.substr(prec_pos, cur_pos - prec_pos)) ;
133 if ((0 != cur.size()) || empty_fields)
134 { result.push_back(cur) ; }
135 prec_pos = cur_pos + size_delim ;
136 cur_pos = input.find(delimiter, cur_pos + size_delim) ;
137 }
138 return result ;
139 }
